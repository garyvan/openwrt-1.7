103 #ifdef HFA_CTX_MEMORY_STATS
104 /**
105  * This macro handle the searchctx memory statistics. It increments the
106  * curent memory usage counter and set the peak and maximum memory counters.
107  */
108 #define hfa_searchctx_mem_stats_inc(ctx, memtype, size) ({ \
109  hfa_cntrs_size_t peaksize = 0; \
110  HFA_CTX_STATS_INC(ctx, memtype.curent, cvmx_get_core_num(), size); \
111  if((HFA_CTX_STATS_GET(ctx, memtype.max, cvmx_get_core_num())) < size) \
112  HFA_CTX_STATS_SET(ctx, memtype.max, cvmx_get_core_num(), size); \
113  peaksize = HFA_CTX_STATS_GET(ctx, memtype.curent, cvmx_get_core_num()); \
114  if((HFA_CTX_STATS_GET(ctx, memtype.peak, cvmx_get_core_num())) < peaksize) \
115  HFA_CTX_STATS_SET(ctx, memtype.peak, cvmx_get_core_num(), peaksize); \
116 })
117 /**
118  * This macro handle the searchctx memory statistics. It decrements the
119  * curent memory usage counter.
120  */
121 #define hfa_searchctx_mem_stats_dec(ctx, memtype, size) \
122 ({ \
123  HFA_CTX_STATS_DEC(ctx, memtype.curent, cvmx_get_core_num(), size); \
124 })
125 #else
126 /* Dummy macros for ctx memory statistics */

135 #ifdef HFA_CORE_MEMORY_STATS
136 /**
137  * This Macro handle the core memory statistics. It increments the
138  * curent memory usage counter and set the peak and maximum memory counters.
139  */
140 #define hfa_core_mem_stats_inc(memtype,size) ({ \
141  int c = cvmx_get_core_num(); \
142  hfa_cntrs_size_t peaksize = 0; \
143  HFA_CORE_STATS_INC(memtype.curent, c, size); \
144  if((HFA_CORE_STATS_GET(memtype.max, c)) < size) \
145  HFA_CORE_STATS_SET(memtype.max, c, size); \
146  peaksize = HFA_CORE_STATS_GET(memtype.curent, c); \
147  if((HFA_CORE_STATS_GET(memtype.peak, c)) < peaksize) \
148  HFA_CORE_STATS_SET(memtype.peak, c, peaksize); \
149 })
150 /**
151  * This macro handle the core memory statistics. It decrements the
152  * curent memory usage counter.
153  */
154 #define hfa_core_mem_stats_dec(memtype, size) \
155 ({ \
156  HFA_CORE_STATS_DEC(memtype.curent, cvmx_get_core_num(), size); \
157 })

159 /* Dummy macros for core memory statistics */
