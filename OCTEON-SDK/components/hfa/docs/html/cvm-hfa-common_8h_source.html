261 /**
262  * Indicates whether recieved wqe is search instruction response.
263  *
264  * @param wqe WQE pointer
265  *
266  * @return 1 if search response, 0 otherwise
267  */
268 static inline hfa_bool_t
269 hfa_search_response(cvmx_wqe_t *wqe)
270 {
271  if(HFA_SEARCH_HWWQE_UNUSED_FIELD == cvmx_wqe_get_unused8(wqe))
272  return (HFA_TRUE);
273  else
274  return (HFA_FALSE);
275 }
276 /**
277  * Indicates whether recieved wqe is graph load instruction response.
278  *
279  * @param wqe WQE pointer
280  *
281  * @return 1 if search response, 0 otherwise
282  */
283 static inline hfa_bool_t
284 hfa_graph_load_response(cvmx_wqe_t *wqe)
285 {
286  if(HFA_GRAPH_HWWQE_UNUSED_FIELD == cvmx_wqe_get_unused8(wqe))
287  return (HFA_TRUE);
288  else
289  return (HFA_FALSE);
290 }
291 /**
292  * This routine is for getting WQE type.
293  *
294  * @param wqe WQE pointer
295  *
296  * @return HFA_SEARCH_HWWQE, if wqe is a search instruction response
297  * HFA_GRAPH_HWWQE, if wqe is a graph load instruction response
298  * PACKET_WQE, if wqe is a packet wqe
299  */
300 static inline hfa_wqe_type_t
301 hfa_get_wqe_type(cvmx_wqe_t *wqe)
302 {
303  switch(cvmx_wqe_get_unused8(wqe)) {
304  case HFA_SEARCH_HWWQE_UNUSED_FIELD:
305  return HFA_SEARCH_HWWQE;
306  case HFA_GRAPH_HWWQE_UNUSED_FIELD:
307  return HFA_GRAPH_HWWQE;
308  default:
309  return PACKET_WQE;
310  }

397 #define WORDLEN 8
398 /**
399  * Dump buffer pointed by "buf" in readable format
400  *
401  * @param msg String to identify dump
402  * @param buf Pointer to the buffer
403  * @param count Size of buffer
404  *
405  * @return Void
406  */
407 static inline void
408 hfa_dump_buf (const char *msg, uint64_t *buf, uint64_t count)
409 {
410  unsigned long i, base = 0;
411  unsigned char *D = (unsigned char *)buf;
412  hfa_log( "Dump: %s, Addr: %p\n", msg, buf);
413  hfa_log( "-------- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- ----------------\n" );
414  for( ; base+WORDLEN < count; base += WORDLEN )
415  {
416  hfa_log( "%08lX ", base );
417  for( i=0; i<WORDLEN; i++ )
418  {
419  hfa_log( "%02X ", D[base+i] );
420  }
421  for( i=0; i<WORDLEN; i++ )
422  {
423  hfa_log( "%c", isprint(D[base+i])?D[base+i]:'.' );
424  }
425  hfa_log( "\n" );
426  }
427  hfa_log( "%08lX ", base );
428  for( i=base; i<count; i++ )
429  {
430  hfa_log( "%02X ", D[i] );
431  }
432  for( ; i<base+WORDLEN; i++ )
433  {
434  hfa_log( " " );
435  }
436  for( i=base; i<count; i++ )
437  {
438  hfa_log( "%c", isprint(D[i])?D[i]:'.' );
439  }
440  for( ; i<base+WORDLEN; i++ )
441  {
442  hfa_log( " ");
443  }
444  hfa_log( "\n" );
445  hfa_log( "-------- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- ----------------\n" );
446  hfa_log( "\n" );
447  }
