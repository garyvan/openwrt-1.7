109 /**

170  * The HFA graph comes with the initial snode in the info section. A partial

187  * saved from cvm_hfa_rword_t::f2 and set to corresponding field in
188  * @ref cvmx_hfa_command_t
189  */
190  uint64_t hash:8;
191  /**
192  * saved from cvm_hfa_rword_t::f1 and set to corresponding field in
193  * @ref cvmx_hfa_command_t
194  */
195  uint64_t ntype:3;
196  uint64_t smdtomtype:1;
197  /**
198  * saved from cvm_hfa_rword_t::nextnode and set to corresponding field
199  * in @ref cvmx_hfa_command_t
200  */
201  uint64_t nextnode:27;
202 #else
203  uint64_t nextnode:27;
204  uint64_t smdtomtype:1;
205  uint64_t ntype:3;
206  uint64_t hash:8;
207  uint64_t dnodeid:9;
208  uint64_t ndnodes:9;
209  uint64_t nbdnodes:6;
210  uint64_t zerobdn:1;
211 #endif
212  } s;
213 } cvm_hfa_snode_t;

233 /**
234  * Initialize the HFA instruction.
235  * Clears @b command to zero and sets its instruction type to @b itype
236  *
237  * @param command pointer to instruction
238  * @param itype instruction type
239  *
240  * @return Zero on success, negative on failure
241  */
242 static inline int cvm_hfa_instr_init(cvmx_hfa_command_t *command,
243  cvm_hfa_itype_t itype)
244 {
245  memset (command, 0, sizeof *command);
246 #ifdef HFA_STRICT_CHECK
247  switch(itype){
248  case CVMX_HFA_ITYPE_GRAPHWALK:
249  hfa_dbg("InstrInit: Search\n");
250  break;
251  case CVMX_HFA_ITYPE_MEMLOAD:
252  hfa_dbg("InstrInit: MemLoad\n");
253  break;
254  case CVMX_HFA_ITYPE_CACHELOAD:
255  hfa_dbg("InstrInit: CacheLoad\n");
256  break;
257  case CVMX_HFA_ITYPE_GRAPHFREE:
258  hfa_dbg("InstrInit: GraphFree\n");
259  break;
260  default:
261  return (-1);
262  }
263 #endif
264  command->word0.itype = itype;
265  return 0;
266 }
