249 /**
250  * This routine is a variant of hfa_searchparam_set_inputiovec() when the
251  * previously searched input IO vectors are also set in sparam.
252  * Only a portion of the input IO vectors will be used to perform the
253  * pattern-search. The @b ioveclen_0_n parameter specifies the total number of
254  * IO vectors including the older IO vectors and newest IO vector yet to be
255  * scanned. The @b ioveclen parameter specifies the number of IO vectors that
256  * need to be scanned(i.e. the Nth set of IO vectors). The IO vectors from @b 0
257  * to <b>[ioveclen_0_n - ioveclen]</b> represent the older(previously searched
258  * IO vectors).
259  *
260  * This routine should be used when hfa_searchparam_get_ppoflags() returns the
261  * HFA_SEARCHCTX_FSAVEBUF flag. This indicates that the post-processor(as part
262  * of hfa_searchctx_getmatches()) may require access to previous input buffers
263  * in order to report some matches. As a result the application must preserve
264  * these buffers and should not free them. hfa_searchctx_getmatches() will make
265  * use of all the IO vectors from @b 0 to @b N.
266  *
267  * If ioveclen_0_n and ioveclen are equal, this routine is effectively same as
268  * hfa_searchparam_set_inputiovec().
269  *
270  * @param psparam Pointer to search parameter
271  * @param piovec_0_n Pointer to iovec array [0..n]
272  * @param ioveclen_0_n Length of total iovecs including nth iovec len
273  * @param ioveclen Length of nth iovec len
274  *
275  * @return HFA_SUCCESS/HFA_FAILURE
276  */
277 static inline hfa_return_t
278 hfa_searchparam_set_inputiovec_0_n (hfa_searchparams_t *psparam,
279  hfa_iovec_t *piovec_0_n, uint32_t ioveclen_0_n, uint32_t ioveclen)
280 {
281  hfa_iovec_t *piovec_n = NULL;
282  uint32_t off=0;
283 
284  if(hfa_os_likely(psparam && piovec_0_n)){
285  if(ioveclen > ioveclen_0_n){
286  return HFA_FAILURE;
287  }
288  HFA_SET (psparam, input_0_n, piovec, piovec_0_n);
289  HFA_SET (psparam, input_0_n, ioveclen, ioveclen_0_n);
290 
291  /*Set input iovec_n*/
292  off = ioveclen_0_n - ioveclen;
293  piovec_n = piovec_0_n + off;
294  HFA_SET (psparam, input_n, piovec, piovec_n);
295  HFA_SET (psparam, input_n, ioveclen, ioveclen);
296 
297  return HFA_SUCCESS;
298  }
299  return HFA_FAILURE;
300 }
301 /**
302  * This routine adds a new input IO vector to Search Param. The Search
303  * Param(sparam) represets a set parameters necessary to perform a pattern
304  * search. It includes the input buffer/payload, the buffer to hold the HFA
305  * engine search results, a match callback setting, a WQE pointer, cluster
306  * setting, and post-processing flags.
307  *
308  * The input IO vector contains chunks of buffer/payload data which will be
309  * searched by HFA engine for patterns. The application must ensure that the
310  * buffers are readable by HFA engine, by using appropriate type of memory.
311  *
312  * Refer to hfa_searchparam_set_inputiovec_0_n() for a variant of this routine
313  * and its use-case.
314  *
315  * @param psparam Pointer to search parameter
316  * @param piovec Pointer to iovec array [0..n]
317  * @param ioveclen Length of iovec array(number of IO vectors)
318  *
319  * @return HFA_SUCCESS/HFA_FAILURE
320  */
321 static inline hfa_return_t
322 hfa_searchparam_set_inputiovec (hfa_searchparams_t *psparam,
323  hfa_iovec_t *piovec, uint32_t ioveclen)
324 {
325  if(hfa_os_likely(psparam && piovec)){
326  HFA_SET (psparam, input_n, piovec, piovec);
327  HFA_SET (psparam, input_n, ioveclen, ioveclen);
328 
329  HFA_SET (psparam, input_0_n, piovec, piovec);
330  HFA_SET (psparam, input_0_n, ioveclen, ioveclen);
331  return HFA_SUCCESS;
332  }
333  return HFA_FAILURE;
334 }

409 /**
410  * This routine retrieves the reason code after a search operation has been
411  * completed by the HFA engine. The reason code is written by the hardware to
412  * the output buffer set in sparam using hfa_searchparam_set_output(). The
413  * routine extracts it from the output buffer and makes it available to the
414  * caller.
415  *
416  * @param psparam Pointer to search parameter
417  * @param preason Address of variable where API writes hardware reason
418  *
419  * @return HFA_SUCCESS/HFA_FAILURE
420  *
421  */
422 static inline hfa_return_t
423 hfa_searchparam_get_hwsearch_reason (hfa_searchparams_t *psparam,
424  hfa_reason_t *preason)
425 {
426  uint64_t *ptr_odata = NULL;
427  uint64_t _rmdata = 0;
428  hfa_rmdata_t *prmdata = NULL;
429  hfa_rptr_overload_t *rbuf = NULL;
430 
431  if(hfa_os_likely(psparam && preason)){
432  rbuf = (hfa_rptr_overload_t *)psparam->output.ptr;
433  ptr_odata = &(rbuf->rptrbase);
434 
435  _rmdata = (((uint64_t *) ptr_odata)[0]);
436  prmdata = ((hfa_rmdata_t *) &_rmdata);
437 
438  psparam->ofields.reason = prmdata->s.reason;
439  *preason = psparam->ofields.reason;
440  return HFA_SUCCESS;
441  }
442  return HFA_FAILURE;
443 }

467 /**
468  * This routine returns the data-buffer offset. The data-buffer offset indicates
469  * how much of the input buffer has been searched by the HFA engine. Ideally the
470  * HFA engine will search the entire buffer and write the results to the output
471  * buffer. However, under some circumstances(for ex: if output buffer is full
472  * and hardware cannot write any more results), the search is incomplete and HFA
473  * engine indicates the progress made using the data-buffer offset.
474  * Application must check this field if HFA engine returned HFA_REASON_RFULL as
475  * the reason code(via hfa_searchparam_get_hwsearch_reason()) and launch another
476  * search after providing a new output buffer. The data-buffer offset should be
477  * used to adjust the input IO vectors such that data that has already been
478  * searched by the hardware is not searched again.
479  *
480  * @param psparam Pointer to search parameter
481  * @param ppdboff Pointer to pdboffset
482  *
483  * @return HFA_SUCCESS/HFA_FAILURE
484  *
485  */
486 static inline hfa_return_t
487 hfa_searchparam_get_hwsearch_pdboff (hfa_searchparams_t *psparam,
488  hfa_pdboff_t *ppdboff)
489 {
490  uint64_t *ptr_odata = NULL;
491  uint64_t _rmdata = 0, _rword = 0;
492  hfa_rmdata_t *prmdata = NULL;
493  hfa_rword_t *prword = NULL;
494  hfa_rptr_overload_t *rbuf = NULL;
495 
496  if(hfa_os_likely(psparam && ppdboff && psparam->output.ptr)){
497  rbuf = (hfa_rptr_overload_t *)psparam->output.ptr;
498  ptr_odata = &(rbuf->rptrbase);
499 
500  _rmdata = (((uint64_t *) ptr_odata)[0]);
501  prmdata = ((hfa_rmdata_t *) &_rmdata);
502  _rword = (((uint64_t *) ptr_odata)[prmdata->s.nument]);
503  prword = (hfa_rword_t *) &_rword;
504 
505  psparam->ofields.pdboffset = prword->s.offset +1;
506  *ppdboff = psparam->ofields.pdboffset;
507  return HFA_SUCCESS;
508  }
509  return HFA_FAILURE;
510 }
