79 /*
80 ppmemcb_t holds the user defined argument used by pp to allocate/free.
81 PP allocates/free memory using os_ppbufalloc(uarg)/os_ppbuffree(uarg)
82 respectively. os_ppbufalloc(uarg)/os_ppbuffree(uarg) are macros defined to
83 function pointers used by pp. By default HFA API sets default function pointers,
84 however application can override function pointer to any user-defined function.
85 
86 @uarg - User agument provided to os_ppbufalloc()/os_ppbufree()
87 */
88 typedef struct {
89  void *uarg;
90 } ppmemcb_t;

204 /*
205  ppmeta_t holds the post processing results metadata
206  nmatch - no of pattern matches
207  allocerr - some memory allocation failed
