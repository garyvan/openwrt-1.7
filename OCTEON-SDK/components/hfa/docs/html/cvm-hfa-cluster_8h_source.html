164 /**
165  * Get Cluster Number for Cluster pointer
166  *
167  * @param pclust Pointer to Cluster
168  * @param pclus_num Sets cluster number in *pclus_num
169  *
170  * @return HFA_SUCCESS if successm FAILURE otherwise
171  */
172 static inline hfa_return_t
173 hfa_cluster_getnumber(hfa_cluster_t *pclust, hfa_size32_t *pclus_num)
174 {
175  if(pclus_num && pclust){
176  *pclus_num = pclust->s.clust_num;
177  return HFA_SUCCESS;
178  }
179  return HFA_FAILURE;
180 }
