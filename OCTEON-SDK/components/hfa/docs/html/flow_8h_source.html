157 /**
158  * Calculate Array index to which received flowid should be present or added
159  *
160  * @param pfdb Pointer to the Flow db
161  * @param id Flow id or Tag value
162  * @param pidx Pointer to the Array Index
163  *
164  * @return void
165  */
166 static inline void
167 hfaf_get_flowidx (hfaf_flowdb_t *pfdb, flowid_t id, uint64_t *pidx)
168 {
169  uint32_t mask = 0x0;
170 
171  mask = id & (pfdb->bitmsk);
172  *pidx = mask >> (pfdb->wqegrp_bits);
173 }
174 /**
175  * Search particular flowid or flow in Flow database and return 1 if present or 0
176  * otherwise. If found, return flow ptr in ppflow
177  *
178  * @param pfdb Pointe to the flowdb
179  * @param id Flowid or Tag value
180  * @param ppflow Pointer to pointer to flow
181  *
182  * @return HFA_SUCCESS if id found otherwise HFA_FAILURE
183  */
184 static inline hfa_return_t
185 hfaf_isflow_exist(hfaf_flowdb_t *pfdb, flowid_t id, hfaf_flow_t **ppflow)
186 {
187  hfa_os_listhead_t *p1 = NULL, *p2 = NULL;
188  hfaf_flowdb_node_t *pdbnode = NULL;
189  hfaf_flowid_ptr_map_t *pmap;
190  uint64_t idx;
191 
192  if(pfdb && ppflow){
193  *ppflow = NULL;
194  hfaf_get_flowidx(pfdb, id, &idx);
195 
196  pdbnode = (hfaf_flowdb_node_t *)&((pfdb->pnodes)[idx]);
197  hfa_os_listforeachsafe(p1, p2, &pdbnode->list){
198  pmap= hfa_os_listentry(p1,hfaf_flowid_ptr_map_t,list);
199  if(id == pmap->flowid){
200  *ppflow = pmap->pflow;
201  goto flowexit;
202  }
203  }
204 flowexit:
205  return HFA_SUCCESS;
206  }
207  return HFA_FAILURE;
208 }
