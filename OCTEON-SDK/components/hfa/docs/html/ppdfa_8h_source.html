145 /*
146  Hash table entry format
147 
148  bucket - Is it a bucket? i.e, no of entries > 1
149  present - Are there any entries at this index?
150  nnptr - size of bucket if bucket is set,
151  else marked node
152  off - offset of bucket/marked node info in
153  info file
154 */
155 typedef union {
156  uint64_t u64;
157  struct {
158 #if __BYTE_ORDER == __BIG_ENDIAN
159  uint64_t off:32;
160  uint64_t nnptr:27;
161  uint64_t unused:3;
162  uint64_t present:1;
163  uint64_t bucket:1;
164 #else
165  uint64_t bucket:1;
166  uint64_t present:1;
167  uint64_t unused:3;
168  uint64_t nnptr:27;
169  uint64_t off:32;
170 #endif
171  } s;
172 } ppdfa_hashent_t;

365 /*
366  ppdfa_snode_t holds start node info of the graph
367 
368  f4 - used in cache load instruction and '1' should be added
369  before filling it, i.e, for cache load,
370  iword3.f4 = f4 + 1
371  dsize - number of ram2 entries
372  f3 - iword3.f3 for first walk instruction in a flow
373  f2 - iword1.f2 for first walk instruction in a flow
374  f1 - iword0.f1 for first walk instruction in a flow
375  small - global small memory flag
376  snode - snode for first walk instruction in a flow
377 */

379  uint64_t u64;
380  struct {
381 #if __BYTE_ORDER == __BIG_ENDIAN
382  uint64_t rsvd0:1;
383  uint64_t f4:6;
384  uint64_t dsize:9;
385  uint64_t f3:9;
386  uint64_t f2:8;
387  uint64_t f1:3;
388  uint64_t small:1;
389  uint64_t snode:27;
390 #else
391  uint64_t snode:27;
392  uint64_t small:1;
393  uint64_t f1:3;
394  uint64_t f2:8;
395  uint64_t f3:9;
396  uint64_t dsize:9;
397  uint64_t f4:6;
398  uint64_t rsvd0:1;
399 #endif
400  } s;
