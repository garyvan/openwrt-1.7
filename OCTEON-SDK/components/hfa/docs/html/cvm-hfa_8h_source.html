154 static inline

171  case OCTEON_HFA_CN63XX_CID:

174  case OCTEON_HFA_CN66XX_CID:

176  break;
177  case OCTEON_HFA_CN68XX_CID:
178  return (HFA_CN68XX_HTES);
179  break;
180  case OCTEON_HFA_CN61XX_CID:
181  return (HFA_CN61XX_HTES);
182  break;
183  case OCTEON_HFA_CN70XX_CID:
184  return (HFA_CN70XX_HTES);
185  break;

189 }

240 /**
241  * Indicates whether device has its own HFA memory. hfa_dev_getdevinfo() should
242  * be called before this routine.
243  *
244  * @param pdev Pointer to device
245  * @return 1 if present, 0 otherwise
246  */
247 static inline hfa_bool_t
248 hfa_dev_haspvt_hfamemory (hfa_dev_t *pdev)
249 {
250  if(hfa_os_likely(pdev)){
251  return (pdev->devinfo.hwhasownmem);
252  } else {
253  switch(OCTEON_HFA_CHIP()){
254  case OCTEON_HFA_CN63XX_CID:
255  case OCTEON_HFA_CN66XX_CID:
256  return(HFA_TRUE);
257  break;
258  default:
259  return(HFA_FALSE);
260  }
261  }
262 }

380 /**
381  * @cond INTERNAL
382  * Returns memory alignment required for OCTEON HFA
383  *
384  * @return hfa_size_t Alignment
385  */
386 static inline hfa_size_t
387 hfa_get_mem_align(void)
388 {
389  uint64_t align;
390 
391  switch(OCTEON_HFA_CHIP()){
392  case OCTEON_HFA_CN68XX_CID:
393  case OCTEON_HFA_CN61XX_CID:
394  case OCTEON_HFA_CN70XX_CID:
395  align = HFA_68XX_MEM_ALIGNMENT;
396  break;
397 
398  case OCTEON_HFA_CN63XX_CID:
399  case OCTEON_HFA_CN66XX_CID:
400  align = HFA_63XX_MEM_ALIGNMENT;
401  break;
402 
403  default:
404  align = HFA_INVALID_ALIGNMENT;
405  }
406  return(align);
407 }
