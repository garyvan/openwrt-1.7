47 #ifndef __CVM_HFA_ERROR_H__

50 /** Max number of errors in uint64_t*/
51 #define CVM_HFA_MAX_ERR 64
52 
53 /**16 bits for General errors*/
54 #define CVM_HFA_GENERR_IDX 1
55 #define CVM_HFA_GEN_ELEN 16
56 
57 /**8 bits for Device errors*/
58 #define CVM_HFA_DEVERR_IDX CVM_HFA_GENERR_IDX + CVM_HFA_GEN_ELEN /**17*/
59 #define CVM_HFA_DEV_ELEN 8
60 
61 /**8 bits for Cluster errors*/
62 #define CVM_HFA_CLUSERR_IDX CVM_HFA_DEVERR_IDX + CVM_HFA_DEV_ELEN /**25*/
63 #define CVM_HFA_CLUS_ELEN 8
64 
65 /**8 bits for Graph errors*/
66 #define CVM_HFA_GRAPHERR_IDX CVM_HFA_CLUSERR_IDX + CVM_HFA_CLUS_ELEN /**33*/
67 #define CVM_HFA_GRAPH_ELEN 8
68 
69 /**Remaining bits for Search errors*/
70 #define CVM_HFA_SRCHERR_IDX CVM_HFA_GRAPHERR_IDX+ CVM_HFA_GRAPH_ELEN /**41*/
71 #define CVM_HFA_SRCH_ELEN CVM_HFA_MAX_ERR - CVM_HFA_SRCHERR_IDX + 1
